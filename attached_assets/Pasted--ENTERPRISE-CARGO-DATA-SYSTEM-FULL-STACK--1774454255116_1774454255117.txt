import express from 'express';
import pkg from 'pg';
import cors from 'cors';
import fs from 'fs';
import dotenv from 'dotenv';
import { Parser } from 'json2csv';

dotenv.config();
const { Client } = pkg;

const app = express();
app.use(cors());
app.use(express.json());

const prod = new Client({ connectionString: process.env.PROD_DB_URL });
const dev = new Client({ connectionString: process.env.DEV_DB_URL });

await prod.connect();
await dev.connect();

// =========================================
// GLOBAL STATE
// =========================================
let syncStatus = { status: 'idle', lastRun: null, rows: 0, error: null };

// =========================================
// GENERIC UPSERT FUNCTION
// =========================================
async function syncTable(table, columns) {
  const lastSync = fs.existsSync('last_sync.txt')
    ? fs.readFileSync('last_sync.txt', 'utf-8')
    : '2000-01-01';

  const res = await prod.query(`SELECT * FROM ${table} WHERE last_updated > $1`, [lastSync]);

  let count = 0;

  for (const row of res.rows) {
    const values = columns.map(c => row[c]);

    const query = `
      INSERT INTO ${table} (${columns.join(',')})
      VALUES (${columns.map((_, i) => `$${i + 1}`).join(',')})
      ON CONFLICT (id) DO UPDATE SET
      ${columns.map(c => `${c}=EXCLUDED.${c}`).join(',')}
    `;

    await dev.query(query, values);
    count++;
  }

  return count;
}

// =========================================
// MAIN SYNC
// =========================================
async function runSync() {
  try {
    syncStatus.status = 'running';
    syncStatus.error = null;

    let total = 0;

    total += await syncTable('airlines', ['id','name','last_updated']);
    total += await syncTable('airports', ['id','name','code','last_updated']);
    total += await syncTable('ground_handlers', ['id','name','last_updated']);
    total += await syncTable('airline_operations', [
      'id','airline_id','airport_id','firms_code','isc_amount','contact_email','contact_number','last_updated'
    ]);

    fs.writeFileSync('last_sync.txt', new Date().toISOString());

    syncStatus = { status: 'success', lastRun: new Date(), rows: total, error: null };

  } catch (e) {
    syncStatus.status = 'error';
    syncStatus.error = e.message;
  }
}

setInterval(runSync, 60000);

// =========================================
// API: SEARCH SYSTEM
// =========================================
app.get('/search', async (req, res) => {
  const { airline, airport } = req.query;

  let query = `
    SELECT a.name AS airline, ap.code AS airport, ao.firms_code,
           ao.isc_amount, ao.contact_email, ao.contact_number
    FROM airline_operations ao
    JOIN airlines a ON ao.airline_id=a.id
    JOIN airports ap ON ao.airport_id=ap.id
    WHERE 1=1
  `;

  const params = [];

  if (airline) {
    params.push(`%${airline}%`);
    query += ` AND a.name ILIKE $${params.length}`;
  }

  if (airport) {
    params.push(`%${airport}%`);
    query += ` AND ap.code ILIKE $${params.length}`;
  }

  const result = await dev.query(query, params);
  res.json(result.rows);
});

// =========================================
// EXPORT CSV
// =========================================
app.get('/export', async (req, res) => {
  const data = await dev.query(`SELECT * FROM airline_operations`);

  const parser = new Parser();
  const csv = parser.parse(data.rows);

  res.header('Content-Type', 'text/csv');
  res.attachment('cargo_data.csv');
  res.send(csv);
});

// =========================================
// STATUS + MANUAL SYNC
// =========================================
app.get('/status', (req, res) => res.json(syncStatus));
app.post('/sync', async (req, res) => { await runSync(); res.json(syncStatus); });

    <h1>🚀 Cargo System</h1>
    <button onclick="sync()">Sync Now</button>
    <button onclick="exportData()">Export CSV</button>

    <h3>Status</h3>
    <pre id="status"></pre>

    <h3>Search</h3>

    <script>
      async function load(){
        const s = await fetch('/status').then(r=>r.json());

      async function search(){
        const airline = document.getElementById('airline').value;

      }

    </script>
  </body>
  </html>
  `);
